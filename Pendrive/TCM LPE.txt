##System Enumeration:
uname -a
cat /proc/version
cat /etc/issue
lscpu
ps aux
ps aux | grep root

##User Enumeration:
whoami
id
sudo -l
cat /etc/passwd
cat /etc/passwd | cut -d ":" -f 1
cat /etc/shadow
cat /etc/group
history

##Network Enumeration:
ifconfig
ip a
ip route
ip neigh
netstat -ano

##Password Hunting:
grep --color=auto -rnw '/' -ie "PASSWORD=" --color=always 2> /dev/null
grep --color=auto -rnw '/' -ie "PASS=" --color=always 2> /dev/null
find . -type f -exec grep -i -I "PASSWORD" {} /dev/null \;
find / -name id_rsa 2> /dev/null
