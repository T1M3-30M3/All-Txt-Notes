Subdomain Finding:

# Using Sublist3r
sublist3r -d example.com -o subdomains.txt

# Using AssetFinder
assetfinder --subs-only example.com | tee -a subdomains.txt

# Using Amass (passive mode)
amass enum -passive -d example.com -o amass_passive.txt

# Using Subfinder
subfinder -d example.com -o subfinder.txt -silent

# Using Amass (active mode)
amass enum -active -d example.com -o amass_active.txt

# Using AltDNS + MassDNS
altdns -i subdomains.txt -o alt_domains.txt -w words.txt
massdns -r resolvers.txt -t A -o S -w massdns_out.txt alt_domains.txt

# Using DNSrecon
dnsrecon -d example.com -t brt -D subdomains-top1mil.txt -c dnsrecon.csv

# Using certspotter
curl -s https://certspotter.com/api/v0/certs\?domain\=example.com | jq -r '.[].dns_names[]' | sort -u

# Using crt.sh
curl -s "https://crt.sh/?q=%.example.com&output=json" | jq -r '.[].name_value' | sort -u

# Using tls.bufferover.run
curl -s "https://tls.bufferover.run/dns?q=example.com" | jq -r .Results[] | cut -d ',' -f3 | sort -u

# Using dnsgen + massdns
dnsgen subdomains.txt | massdns -r resolvers.txt -t A -o S -w dnsgen_out.txt

# Using puredns
puredns bruteforce all.txt example.com -r resolvers.txt -w puredns_out.txt

# Using gobuster
gobuster dns -d example.com -w subdomains-top1mil.txt -t 50 -o gobuster_out.txt

# Using Waybackurls
waybackurls example.com | unfurl domains

# Using gau
gau --subs example.com | unfurl domains

# Using CommonCrawl
python3 cc.py example.com

JS:

# Get all Wayback URLs for your domain
waybackurls example.com > wayback_urls.txt

# Alternative using gau (gets more sources)
gau example.com > gau_urls.txt

# Combine and filter for JS files
cat wayback_urls.txt gau_urls.txt | sort -u | grep -E "\.js($|\?)" > js_files.txt

# Download all JS files
mkdir js_files
cat js_files.txt | while read url; do
    filename=$(echo "$url" | sed 's/[^a-zA-Z0-9]/_/g').js
    curl -s "$url" -o "js_files/$filename"
done

# Search for API keys, tokens, etc.
grep -E -i -r "api[^a-zA-Z0-9]|token[^a-zA-Z0-9]|key[^a-zA-Z0-9]|secret[^a-zA-Z0-9]|auth[^a-zA-Z0-9]|cred[^a-zA-Z0-9]|password[^a-zA-Z0-9]" js_files/

# Search for hardcoded credentials
grep -E -i -r "username|user|login|pass|pwd|credential|access[^a-zA-Z0-9]key|secret[^a-zA-Z0-9]key" js_files/

# Search for interesting endpoints
grep -E -i -r "/api/|/v1/|/v2/|/internal/|/admin/|/private/|/auth/|/oauth/" js_files/
