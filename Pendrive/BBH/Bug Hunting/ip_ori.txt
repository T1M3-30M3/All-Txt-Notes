#!/bin/bash

# API keys (replace with your own keys)
VT_API_KEY="YOUR_VT_API_KEY_HERE"

# Function to fetch IP addresses from VirusTotal
fetch_vt_ips() {
    local domain=$1
    response=$(curl -s "https://www.virustotal.com/vtapi/v2/domain/report?domain=$domain&apikey=$VT_API_KEY")
    
    # Check for errors in the response
    if echo "$response" | grep -q '"response_code": 0'; then
        echo "Error fetching data from VirusTotal: $(echo "$response" | jq -r '.verbose_msg')"
        return
    fi

    echo "$response" | jq -r '.. | .ip_address? // empty' | grep -Eo '([0-9]{1,3}\.){3}[0-9]{1,3}'
}

# Function to fetch IP addresses from AlienVault
fetch_otx_ips() {
    local domain=$1
    response=$(curl -s "https://otx.alienvault.com/api/v1/indicators/hostname/$domain/url_list?limit=500&page=1")
    
    # Check for errors in the response
    if echo "$response" | grep -q '"error":'; then
        echo "Error fetching data from AlienVault: $(echo "$response" | jq -r '.error')"
        return
    fi

    echo "$response" | jq -r '..url_list|.[].result?.urlworker?.ip // empty' | grep -Eo '([0-9]{1,3}\.){3}[0-9]{1,3}'
}

# Check if domain is provided
if [ -z "$1" ]; then
    echo "Usage: $0 <domain_name_or_url>"
    exit 1
fi

DOMAIN=$1
OUTPUT_FILE="${DOMAIN}_ips.txt"

# Get IPs from both sources, remove duplicates, and save to file
echo "Collecting IP addresses for: $DOMAIN"
{
    fetch_vt_ips $DOMAIN
    fetch_otx_ips $DOMAIN
} | sort -u | tee "$OUTPUT_FILE"

echo "-----------------------------------"
echo "IP addresses saved to: $OUTPUT_FILE"
